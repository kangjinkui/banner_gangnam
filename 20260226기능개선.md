  🎯 권장 설계 방안: 단일 테이블 확장 (Single Table with Type Discriminator)

  ✅ 선택 이유

  1. 높은 코드 재사용성: 현수막의 핵심 속성이 90% 동일 (주소, 좌표, 기간, 문구, 사진, 메모)
  2. 기존 인프라 활용: 지도, 목록, 통계 로직을 그대로 재사용 가능
  3. API 통합 용이: 필터링만으로 타입별 조회 가능
  4. 유지보수 효율: 버그 수정 시 한 곳만 수정
  5. 확장성: 향후 다른 현수막 타입 추가 시 컬럼만 추가하면 됨

  1. Banner 테이블 스키마 변경

  -- 기존 banners 테이블에 컬럼 추가
  ALTER TABLE banners
    -- 현수막 타입 추가 (정치/공공/집회)
    ADD COLUMN banner_type TEXT NOT NULL DEFAULT 'political'
      CHECK (banner_type IN ('political', 'public', 'rally')),

    -- 공공현수막 전용: 부서명
    ADD COLUMN department TEXT,

    -- party_id를 nullable로 변경 (공공/집회는 정당 불필요)
    ALTER COLUMN party_id DROP NOT NULL,

    -- 기간을 선택입력으로 (nullable)
    ALTER COLUMN start_date DROP NOT NULL,
    ALTER COLUMN end_date DROP NOT NULL;

  -- 제약조건 수정: 정치 현수막만 party_id 필수
  ALTER TABLE banners
    ADD CONSTRAINT check_political_party
      CHECK (
        (banner_type = 'political' AND party_id IS NOT NULL)
        OR
        (banner_type IN ('public', 'rally') AND party_id IS NULL)
      );

  -- 제약조건 추가: 공공 현수막만 department 필수
  ALTER TABLE banners
    ADD CONSTRAINT check_public_department
      CHECK (
        (banner_type = 'public' AND department IS NOT NULL)
        OR
        (banner_type IN ('political', 'rally') AND department IS NULL)
      );

  -- 기간 validation 수정 (nullable 허용)
  ALTER TABLE banners
    DROP CONSTRAINT IF EXISTS check_date_range,
    ADD CONSTRAINT check_date_range
      CHECK (
        (end_date IS NULL)
        OR
        (start_date IS NOT NULL AND end_date >= start_date)
      );

  -- 인덱스 추가
  CREATE INDEX idx_banners_type ON banners(banner_type);
  CREATE INDEX idx_banners_department ON banners(department);

  🔌 API 설계

  1. 기존 API 엔드포인트 확장

  변경 없음: 기존 /api/banners 엔드포인트를 그대로 사용하되, banner_type 필터 추가

  GET /api/banners

  신규 Query Parameters:
  banner_type?: 'political' | 'public' | 'rally' | 'all'  // 기본값: 'political'
  department?: string                                     // 부서명 필터
  exclude_rally_expired?: boolean                         // 집회 현수막은 만료 제외 (기본: false)        

  사용 예시:
  - 공공 현수막만: GET /api/banners?banner_type=public
  - 집회 현수막만: GET /api/banners?banner_type=rally
  - 공공+집회: GET /api/banners?banner_type=public,rally
  - 만료된 공공 현수막: GET /api/banners?banner_type=public&status=expired

  POST /api/banners

  신규 Required Fields:
  {
    banner_type: 'public' | 'rally',  // 필수
    department?: string,              // banner_type='public'일 때 필수
    party_id: null,                   // 공공/집회는 null
    start_date?: string,              // 선택
    end_date?: string,                // 선택
    // ... 기존 필드
  }

  GET /api/banners/stats

  신규 Response Fields:
  {
    // 기존 stats
    total: number;
    active: number;
    expired: number;

    // 신규 stats
    by_type: {
      political: number;
      public: number;
      rally: number;
    };
    by_department: {
      [department: string]: {
        total: number;
        active: number;
        expired: number;  // 공공만 해당
      };
    };
  }

  2. 만료 기능 차별화 로직

  // Service Layer에서 처리
  function isExpired(banner: Banner): boolean {
    // 집회 현수막은 만료 개념 없음
    if (banner.banner_type === 'rally') {
      return false;
    }

    // 공공/정치 현수막은 end_date 기준 만료 판정
    if (!banner.end_date) {
      return false;
    }

    return new Date(banner.end_date) < new Date();
  }
